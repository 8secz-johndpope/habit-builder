Usage: habit-builder [arguments]

  --password=password                   User's password
  --json=/path/to/file.json             Path to firebaseAdminSDK.json
  --help                                Show this message
  --list                                Show Habits list of currentUser
  --add $habitName                      Add new Habit
  --habit=$habit --show                 Show the $habit
  --habit=$habit --del                  Delete $habit
  --habit=$habit --tick                 I did it!
  --habit=$habit --tick -days=x         I did it x days ago
  --habit=$habit --remove=$logKey       Remove $logKey of $habit habit
